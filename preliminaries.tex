\headerbox{Preliminaries}{name=preliminaries,column=1,row=0}{
\begin{minipage}[t]{0.32\textwidth}
\textbf{Federated Learning (FL):}
Collaborative training where clients train local models and share updates with a central server.
\end{minipage}
\hfill
\begin{minipage}[t]{0.32\textwidth}
\textbf{Decentralized FL:}
Peers communicate directly via peer-to-peer networks, eliminating central points of failure.
\end{minipage}
\hfill
\begin{minipage}[t]{0.32\textwidth}
\textbf{Secure Aggregation:}
Cryptographic protocol ensuring only aggregated updates are revealed, protecting individual contributions.
\end{minipage}

\vspace{0.5em}

\begin{minipage}[t]{0.49\textwidth}
\textbf{IPFS:}
Distributed file system using Content Identifiers (CID) for immutable storage and retrieval.
\end{minipage}
\hfill
\begin{minipage}[t]{0.49\textwidth}
\textbf{Hyperledger Fabric:}
Permissioned blockchain for smart contracts, enabling auditable model provenance and authorization.
\end{minipage}
}
